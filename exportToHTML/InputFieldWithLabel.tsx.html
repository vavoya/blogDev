import styles from "./InputFieldWithLabel.module.css"

export interface InputFieldWithLabelProps {
    labelText: string
    errorText?: string
    placeholder?: string
    value?: string
    readOnly?: boolean
    onChange?: (e: React.ChangeEvent<HTMLInputElement>) => void
    id?: number
}

export default function InputFieldWithLabel({id, labelText, errorText, placeholder, value = undefined, readOnly = false, onChange = () => null}: InputFieldWithLabelProps) {
    return (
        <div className={styles.inputBox}>
            <label>
                {labelText}
            </label>
            <input
                key={id}
                defaultValue={value}
                onChange={onChange}
                readOnly={readOnly}
                placeholder={placeholder}/>
            {
                errorText &&
                <span>
                    {errorText}
                </span>
            }
        </div>
    )
}
